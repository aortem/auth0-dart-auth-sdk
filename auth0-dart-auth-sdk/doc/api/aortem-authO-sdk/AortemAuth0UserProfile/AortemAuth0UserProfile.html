Future<AortemAuth0UserProfileResponse> aortemAuth0UserProfile(
  AortemAuth0UserProfileRequest request,
  String auth0Domain, // The Auth0 domain (e.g., 'your-domain.auth0.com')
) async {
  final url = Uri.https(auth0Domain, '/userinfo');

  // Ensure that accessToken is provided
  if (request.accessToken == null || request.accessToken.isEmpty) {
    throw ArgumentError('Access token is required');
  }

  // Send GET request to the /userinfo endpoint with Authorization header
  final response = await http.get(
    url,
    headers: {
      'Content-Type': 'application/json',
      'Authorization': 'Bearer ${request.accessToken}',
    },
  );

  if (response.statusCode == 200) {
    // Successfully retrieved user profile, parse the response
    try {
      final responseBody = json.decode(response.body);
      return AortemAuth0UserProfileResponse.fromJson(responseBody);
    } catch (e) {
      throw AortemAuth0UserProfileException('Error parsing response: $e');
    }
  } else {
    // Handle error responses
    final errorMessage = _getErrorMessage(response);
    throw AortemAuth0UserProfileException(errorMessage);
  }
}
